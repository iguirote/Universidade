B) Problema das Jarras

jr1 suporta 4L
jr2 suporta 3L
torneira que jorra água infinitamente

objetivo : Deixar em uma das jarras 2L

class Estado (int j1, int j2) {
    this.j1 = j1;
    this.j2 = j2;
}

public boolean ehMeta () {
    return ((this.j1 == 2 && this.j2 == 0) || (this.j1 == 0 && this.j2 == 2));
}

//Encher Jarras
public void encher j1 () {
    this.j1 = 4;
}
public void encher j2 () {
    this.j2 = 3;
}

//Esvaziar Jarras
public void esvaziar j1 () {
    this.j1 = 0;
}
public void esvaziar j2 () {
    this.j2 = 0;
}

//Despejar jarras 
public void despejar j1_j2() {
    int j1 = this.j1 = 1;
    int j2 = this.j2 = 2;

    if(j1 > (3-j2)){
        j1 = j1 - (3 - j2);
        j2 = 3;
    } else {
        j2 = j1-j2;
        j1 = 0;
    }
}

public void despejar j2_j1() {
    int j1 = this.j1 = 1;
    int j2 = this.j2 = 2;

    if(j2 > (4-j1)){
        j2 = j2 - (4 - j1);
        j1 = 4;
    } else {
        j1 = j1-j2;
        j2 = 0;
    }
}
